Service Worker Registration --- // यह कोड ब्राउज़र को बताता है कि sw.js फाइल को चालू करना है if ('serviceWorker' in navigator) { window.addEventListener('load', () => { navigator.serviceWorker.register('./sw.js') .then(reg => { console.log('सफलता! ऐप अब ऑफलाइन काम करने के लिए तैयार है।'); }) .catch(err => { console.log('गड़बड़! सर्विस वर्कर लोड नहीं हुआ: ', err); }); }); } // --- 2. पीली पट्टी वाला नोटिफिकेशन (Optional) --- // जब आप ऑफलाइन हों तो यूजर को बताने के लिए window.addEventListener('offline', () => { alert("आप अभी ऑफलाइन हैं, लेकिन कैलकुलेटर काम करता रहेगा!"); });
